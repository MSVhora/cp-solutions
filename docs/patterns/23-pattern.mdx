## Problem Statement

Print the given pattern.

```text

   *     *
  * *   * *
 *   * *   *
*     *     *

```

## Approach

- In the given pattern, we can see there are **n** rows and **4n - 3** columns.
- We have to print starting spaces i.e. **n - rowIndex** number of spaces for each row
- Then we have to print hollow pyramid pattern, for that we will traverse from **1** to **2 * rowIndex - 1** and print **\*** on edges only else space.
- We have print middle space using **2 * (n - rowIndex) - 1**
- We will repeat the hollow pyramid to complete the pattern.
- Note: Take care of last row to not include double edges on last rowIndex.

| :---: | :---: | :---: | :---: | :---: | :---: | :---: | :---: | :---: | :---: | :---: | :---: | :---: | :---: |
|   1   |       |       |       |   *   |       |       |       |       |       |   *   |       |       |       |
|   2   |       |       |   *   |       |   *   |       |       |       |   *   |       |   *   |       |       |
|   3   |       |   *   |       |       |       |   *   |       |   *   |       |       |       |   *   |       |
|   4   |   *   |       |       |       |       |       |   *   |       |       |       |       |       |   *   |

```kt
fun pattern23(n: Int) {
    for(rowIndex in 1..n) {
        val noOfCols = 2 * rowIndex - 1
        val noOfSpaces = n - rowIndex
        
        // Print starting spaces
        print(" ".repeat(noOfSpaces))

        // print first wave character
        for(colIndex in 1..noOfCols) {
            if(colIndex == 1 || colIndex == noOfCols)
                print("*")
            else
                print(" ")
        }

        if(rowIndex != n) {
            // Print middle spaces
            print(" ".repeat(2 * noOfSpaces - 1))

            // print first wave character
            for(colIndex in 1..noOfCols) {
                if(colIndex == 1 || colIndex == noOfCols)
                    print("*")
                else
                    print(" ")
            }
        } else {
            print(" ".repeat(2 * n - 3))
            print("*")
        }
        println()
    }
}

fun main() {    
    pattern23(4)
}
```
