## Problem Statement

Print the given pattern.

```text

1 
0 1 
1 0 1 
0 1 0 1 
1 0 1 0 1 

```
## Approach

- In the given pattern, we can see there are **n** rows and **n** columns.
- We have to print values **0** and **1** in alternate pattern for columns as well as rows
- From table, we can see we have to print **rowIndex** number of columns in each row

<TableWrapper rows="5" cols="5">

|   	| 1 	| 2 	| 3 	| 4 	| 5 	|
|:-:	|:-:	|:-:	|:-:	|:-:	|:-:	|
| 1 	| 1 	|   	|   	|   	|   	|
| 2 	| 0 	| 1 	|   	|   	|   	|
| 3 	| 1 	| 0 	| 1 	|   	|   	|
| 4 	| 0 	| 1 	| 0 	| 1 	|   	|
| 5 	| 1 	| 0 	| 1 	| 0 	| 1 	|

</TableWrapper>

```kt
fun pattern22(n: Int) {
    var rowIndexValue = 1
    for(rowIndex in 1..n) {
        var colValue = rowIndexValue
        for (col in 1..rowIndex) {
            print("$colValue ")
            colValue = if(colValue == 1) 0 else 1 
        }
        rowIndexValue = if(rowIndexValue == 1) 0 else 1
        println()
    }
}

fun main() {	
    pattern22(5)
}
```
